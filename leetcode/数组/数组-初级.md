#### 旋转数组问题：
【数组平移：环形移动解法】
```
class Solution {
public:
    void rotate(vector<int>& nums, int k) {
        k%=(nums.size());
        //int tmp=0;
        ///int start=0;
        int count=0;
        for(int i=0;count<nums.size();i++)
        {
            int current=i;
            int pred=nums[current];
            while(true)
                {
                    int next=(current+k)%nums.size();
                    int tmp=nums[next];
                    nums[next]=pred;
                    pred=tmp;
                    current=next;
                    //j=(j+k)%nums.size();
                    count++;
                    if(i==current){
                        break;
                    }
                }

        }     
        }
};
```

class Solution {

        //int tmp=0;
        ///int start=0;
        int tmp=0;
        for(int i=0;i<k;i++)
        {
            for(int j=0;j<nums.size()-1;j++){
                
                if(j==0){
                    tmp=nums[j+1];
                nums[j+1]=nums[j];
                }
                else{
                    int t=nums[j+1];
                    nums[j+1]=tmp;
                    tmp=t;

                }
                }
                nums[0]=tmp;
        }
        }
};
```

####【反转数组】
```
class Solution {
public:
    void reverse(vector<int>& nums, int start,int end){
        while(start<end){
            int tmp=nums[start];
            nums[start]=nums[end];
            nums[end]=tmp;
            start++;
            end--;
        }
    }
    void rotate(vector<int>& nums, int k) {
        k%=(nums.size());
        reverse(nums,0,nums.size()-1);
        reverse(nums,0,k-1);
        reverse(nums,k,nums.size()-1)；  
        }
};
```

####【数组中存在重复元素】


利用map等哈希结构进行搜索
```
class Solution {
public:
    bool containsDuplicate(vector<int>& nums) {
        unordered_map<int,int> keys;
        for(int i=0;i<nums.size();i++){
            keys[nums[i]]++;
            if(keys[nums[i]]>1)
                return true;
            
        }
        return false;
    }
};
